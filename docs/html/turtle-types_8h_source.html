176// Im Syntaxbaum werden folgende Knotentypen verwendet,
177// mit folgenden Unter-Syntaxb‰umen im son-Array (keine wenn nichts angegeben)
178// und folgender Information im Art-abh‰ngigen Member d
179//
180// name_any Variable (lesend in einer Rechnung): d = Namenseintrag (p_name)
181// oper_const Zahl: Konstante, d = Wert (val)
182// oper_neg un‰res -: son 0 = Operand
183// oper_pow, oper_mul, oper_div, oper_add, oper_sub
184// ^ * / + -: son 0 und son 1 = Operanden
185// oper_abs, Absolutbetrag: son 0 = Operand
186// oper_lpar '(' = Funktionsaufruf:
187// d = Namenseintrag der Funktion (p_name)
188// son 0, 1, ...: Argumente (nicht vorhandene Argumente sind NULL)
189// oper_equ, oper_nequ, oper_less, oper_lequ, oper_grtr, oper_gequ
190// = <> < <= > >=: son 0 und son 1 = Operanden
191// oper_not Logisches not: son 0 = Operand
192// oper_and, oper_or Logisches and und or: son 0 und son 1 = Operanden
193// keyw_walk, keyw_jump
194// walk oder jump in allen Kombinationen
195// d = Art des walk: keyw_walk, keyw_back, keyw_home, keyw_mark
196// bei walk/jump und walk/jump back: son 0 = expr
197// keyw_right turn und turn right: son 0 = expr
198// keyw_left turn left: son 0 = expr
199// keyw_direction direction: son 0 = expr
200// keyw_clear, keyw_stop, keyw_finish, keyw_mark
201// clear, stop, finish, mark: Keine Operanden
202// keyw_path path-Aufruf: d = Name der Pfadfunktion, son 0, 1, ...: Argumente
203// nicht vorhandene Argumente sind NULL
204// keyw_color color: son 0 = rot, son 1 = gr¸n, son 2 = blau
205// keyw_store, keyw_add, keyw_sub, keyw_mul, keyw_div
206// store, add, sub, mul, div:
207// d = Namens-Eintrag der Variable (p_name),
208// son 0 = expr
209// keyw_if if: son 0 = cond, son 1 = then-statements,
210// son 2 = else-statements oder NULL
211// keyw_do, times-Schleife: son 0 = expr, son 1 = statements
212// keyw_counter counter-Schleife:
213// d = Namens-Eintrag der Variable (p_name),
214// son 0 = Startwert
215// to: son 1 = Endwert, son 2 = NULL
216// downto: son 1 = NULL, son 2 = Endwert
217// son 3 = Schrittwert oder NULL
218// son 4 = statements
219// keyw_while while-Schleife: son 0 = cond, son 1 = statements
220// keyw_repeat repeat-Schleife: son 0 = cond, son 1 = statements
221//
222// Vorrang-Klammern in Rechnungen und Bedingungen
223// sind *keine* eigenen Knoten im Syntaxbaum
224// sondern beeinflussen beim Parsen nur die Struktur des Syntaxbaumes
225
226// Typ eines Syntaxbaum-Knotens
227struct _treenode {
228 type_t type; // Art des Knotens
229 srcpos_t pos; // Position im Sourcefile (f¸r Fehlermeldungen)
230 treenode_t *next; // Bei allen Statements: N‰chstes Statement
231 // (oder NULL beim letzten Statement einer Statement-Liste)
232 // Bei allen anderen Knoten: NULL
233 treenode_t *son[MAX_ARGS]; // Sohn-Syntaxb‰ume
234 size_t son_len; // Anzahl an Sohn-Syntaxb‰umen
235 nodedata_t d; // Daten zum Knoten
236};
